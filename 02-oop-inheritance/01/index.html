Setting up movie events (same class as listener Ex02)...

lordOfTheRings = new Movie("The lord of the rings", 2000, 60);

lordOfTheRings.on('play', (movie) => {
console.log(`${movie.name}: is now playing...`);
});

lordOfTheRings.on('pause', (movie) => {
console.log(`${movie.name}: has been paused...`);
});

lordOfTheRings.on('resume', (movie) => {
console.log(`${movie.name}: resumed playing...`);
});

lordOfTheRings.on('finish', (movie) => {
console.log(`${movie.name}: has finished playing...`);
});

lordOfTheRings.on('playTick', (movie) => {
console.log(`${movie.name}: ${movie.status.playTime.toTimeString().substring(0, 8)}`);
});


Setting up movie events (third party class as listener Ex03)...

lordOfTheRings = new Movie("The lord of the rings", 2000, 60);
log = new Logger();

lordOfTheRings.on('play', (movie) => {
log.log(`${movie.name} - play`);
});

lordOfTheRings.on('pause', (movie) => {
log.log(`${movie.name} - pause`);
});

lordOfTheRings.on('resume', (movie) => {
log.log(`${movie.name} - resume`);
});

lordOfTheRings.on('finish', (movie) => {
log.log(`${movie.name} - finish`);
});

lordOfTheRings.on('playTick', (movie) => {
log.log(`${movie.name} - playing(${movie.status.playTime.toTimeString().substring(0, 8)})`);
});
